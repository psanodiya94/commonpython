9class CommonPythonError(Exception):
10 """
11 Base exception class for all CommonPython framework errors.
12
13 @brief Base exception for CommonPython framework.
14 """

16 def __init__(self, message: str, details: dict = None):
17 """
18 Initialize the exception.
19
20 @brief Initialize exception with message and optional details.
21 @param message Error message
22 @param details Optional dictionary with additional error context
23 """
24 self.message = message
25 self.details = details or {}
26 super().__init__(self.message)
27
28 def __str__(self) -> str:
29 """String representation of the exception."""
30 if self.details:
31 details_str = ", ".join(f"{k}={v}" for k, v in self.details.items())
32 return f"{self.message} ({details_str})"
33 return self.message

344def map_exception(original_exception: Exception, message: str = None) -> CommonPythonError:
345 """
346 Map standard Python exceptions to CommonPython custom exceptions.
347
348 @brief Convert standard exceptions to custom exceptions.
349 @param original_exception Original exception instance
350 @param message Optional custom message
351 @return Mapped CommonPython exception
352
353 Example:
354 >>> try:
355 ... open('missing.txt')
356 ... except FileNotFoundError as e:
357 ... raise map_exception(e, "Configuration file not found")
358 """
359 exc_type = type(original_exception).__name__
360 custom_exc_class = EXCEPTION_MAP.get(exc_type, CommonPythonError)
361
362 error_message = message or str(original_exception)
363 return custom_exc_class(error_message, details={"original_error": str(original_exception)})
