29 @staticmethod
30 def create_database_manager(config: Dict[str, Any], logger=None) -> IDatabaseManager:
31 """
32 Create database manager based on configuration.
33
34 @brief Create database manager with automatic implementation selection.
35 @param config Database configuration dictionary
36 @param logger Logger manager instance (optional)
37 @return IDatabaseManager implementation instance
38 @throws ValueError If specified implementation is not available
39 @throws ImportError If required dependencies are missing
40
41 Configuration keys:
42 - implementation: 'cli' or 'library' (default: 'cli')
43 - auto_fallback: True to automatically fallback to CLI if library unavailable (default: True)
44 """
45 impl_type = config.get('implementation', 'cli').lower()
46 auto_fallback = config.get('auto_fallback', True)
47
48 # Check library adapter availability (cached)
49 if ManagerFactory._adapter_cache['db2_library_available'] is None:
50 try:
51 from ..adapters.db2_library_adapter import DB2LibraryAdapter, HAS_IBM_DB
52 ManagerFactory._adapter_cache['db2_library_available'] = HAS_IBM_DB
53 except ImportError:
54 ManagerFactory._adapter_cache['db2_library_available'] = False
55
56 if impl_type == 'library':
57 if ManagerFactory._adapter_cache['db2_library_available']:
58 from ..adapters.db2_library_adapter import DB2LibraryAdapter
59 if logger:
60 logger.logger.info("Using DB2 library adapter (ibm_db)")
61 return DB2LibraryAdapter(config, logger)
62 elif auto_fallback:
63 if logger:
64 logger.logger.warning(
65 "DB2 library adapter not available (ibm_db not installed), "
66 "falling back to CLI adapter"
67 )
68 from ..adapters.db2_cli_adapter import DB2CLIAdapter
69 return DB2CLIAdapter(config, logger)
70 else:
71 raise ValueError(
72 "DB2 library implementation requested but ibm_db is not installed. "
73 "Install with: pip install ibm_db"
74 )
75
76 elif impl_type == 'cli':
77 from ..adapters.db2_cli_adapter import DB2CLIAdapter
78 if logger:
79 logger.logger.info("Using DB2 CLI adapter")
80 return DB2CLIAdapter(config, logger)
81
82 else:
83 raise ValueError(
84 f"Unknown database implementation type: '{impl_type}'. "
85 f"Valid options are 'cli' or 'library'"
86 )
