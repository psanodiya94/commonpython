52 def _build_connection_string(self) -> str:
53 """
54 Build DB2 connection string from configuration.
55
56 @brief Create connection string for ibm_db.
57 @return Formatted connection string
58 """
59 host = self._config.get('host', 'localhost')
60 port = self._config.get('port', 50000)
61 database = self._config.get('database', self._config.get('name', 'testdb'))
62 user = self._config.get('user', 'db2inst1')
63 password = self._config.get('password', '')
64 schema = self._config.get('schema', '')
65
66 conn_str = f"DATABASE={database};HOSTNAME={host};PORT={port};"
67 conn_str += f"PROTOCOL=TCPIP;UID={user};PWD={password};"
68
69 if schema:
70 conn_str += f"CURRENTSCHEMA={schema};"
71
72 return conn_str

132 def execute_query(self, query: str, params: Optional[Tuple] = None) -> List[Dict[str, Any]]:
133 """
134 Execute a SELECT query and return results.
135
136 @brief Execute SELECT query via ibm_db with parameterization support.
137 @param query SQL query string (use ? for parameters)
138 @param params Query parameters tuple
139 @return List of dictionaries containing query results
140 @throws Exception If query execution fails
141 """
142 if not self.is_connected():
143 raise Exception("Database connection not established")
144
145 start_time = time.time()
146 results = []
147
148 try:
149 cursor = self._dbi_conn.cursor()
150
151 if params:
152 cursor.execute(query, params)
153 else:
154 cursor.execute(query)
155
156 # Get column names
157 columns = [desc[0] for desc in cursor.description] if cursor.description else []
158
159 # Fetch all rows
160 for row in cursor.fetchall():
161 result_dict = {}
162 for i, value in enumerate(row):
163 if i < len(columns):
164 result_dict[columns[i]] = value
165 results.append(result_dict)
166
167 cursor.close()
168
169 duration = time.time() - start_time
170 if self._logger:
171 self._logger.log_database_operation(
172 operation="SELECT",
173 query=query,
174 duration=duration,
175 rows_affected=len(results)
176 )
177
178 return results
179
180 except Exception as e:
181 duration = time.time() - start_time
182 if self._logger:
183 self._logger.logger.error(f"Query execution error: {str(e)}")
184 self._logger.log_database_operation(
185 operation="SELECT",
186 query=query,
187 duration=duration,
188 rows_affected=0
189 )
190 raise

245 def execute_batch(self, queries: List[str], params_list: Optional[List[Tuple]] = None) -> List[int]:
246 """
247 Execute multiple queries in a batch.
248
249 @brief Execute batch queries via ibm_db within a transaction.
250 @param queries List of SQL query strings
251 @param params_list List of parameter tuples for each query
252 @return List of affected rows for each query
253 @throws Exception If batch execution fails
254 """
255 if not self.is_connected():
256 raise Exception("Database connection not established")
257
258 results = []
259
260 try:
261 # Begin transaction
262 cursor = self._dbi_conn.cursor()
263
264 for i, query in enumerate(queries):
265 params = params_list[i] if params_list and i < len(params_list) else None
266
267 if params:
268 cursor.execute(query, params)
269 else:
270 cursor.execute(query)
271
272 results.append(cursor.rowcount)
273
274 cursor.close()
275
276 # Commit transaction
277 self._dbi_conn.commit()
278
279 if self._logger:
280 self._logger.logger.info(f"Batch execution completed: {len(queries)} queries")
281
282 except Exception as e:
283 # Rollback transaction
284 self._dbi_conn.rollback()
285 if self._logger:
286 self._logger.logger.error(f"Batch execution error: {str(e)}")
287 raise
288
289 return results

291 @contextmanager
292 def transaction(self):
293 """
294 Context manager for database transactions.
295
296 @brief Provide transaction context via ibm_db.
297 @throws Exception If transaction management fails
298 """
299 if not self.is_connected():
300 raise Exception("Database connection not established")
301
302 try:
303 # ibm_db_dbi uses auto-commit by default, we disable it for transaction
304 if self._logger:
305 self._logger.logger.debug("Transaction started")
306
307 yield self
308
309 # Commit transaction
310 self._dbi_conn.commit()
311 if self._logger:
312 self._logger.logger.debug("Transaction committed")
313
314 except Exception as e:
315 # Rollback transaction
316 self._dbi_conn.rollback()
317 if self._logger:
318 self._logger.logger.error(f"Transaction rolled back: {str(e)}")
319 raise
