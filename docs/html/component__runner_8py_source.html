74 def run(self, args: Optional[list] = None) -> bool:
75 """
76 Run the component.
77
78 @brief Execute component with command-line arguments.
79 @param args Command-line arguments (None to use sys.argv)
80 @return True if component executed successfully, False otherwise
81 """
82 try:
83 # Parse arguments
84 parser = self.create_parser()
85 parsed_args = parser.parse_args(args)
86
87 # Create component instance
88 self.component_instance = self.component_class(
89 parsed_args.config
90 )
91
92 # Set log level if specified
93 if parsed_args.log_level:
94 self.component_instance.logger_manager.set_level(parsed_args.log_level)
95
96 # Set verbose mode
97 if parsed_args.verbose:
98 self.component_instance.logger_manager.set_level('DEBUG')
99
100 # Set dry-run mode
101 if parsed_args.dry_run:
102 self.component_instance.set_config('dry_run', True)
103 self.component_instance.log_info("Running in dry-run mode")
104
105 # Start component
106 return self.component_instance.start()
107
108 except KeyboardInterrupt:
109 if self.component_instance:
110 self.component_instance.log_info("Component interrupted by user")
111 return False
112
113 except Exception as e:
114 if self.component_instance:
115 self.component_instance.log_error(f"Component runner error: {str(e)}")
116 else:
117 print(f"Component runner error: {str(e)}")
118 return False
