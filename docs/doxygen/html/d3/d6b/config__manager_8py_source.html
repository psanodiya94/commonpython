41 def _load_config(self) -> None:
42 """
43 Load configuration from file and environment variables. Only fail if config_file is provided and missing.
44 """
45 if self._config_file:
46 if not os.path.exists(self._config_file):
47 raise FileNotFoundError(f"Configuration file '{self._config_file}' not found.")
48 with open(self._config_file, 'r', encoding='utf-8') as file:
49 self._config = yaml.safe_load(file) or {}
50 self._load_from_env()

78 def _set_nested_value(self, path: str, value: Any) -> None:
79 """
80 Set a nested configuration value using dot notation.
81
82 @brief Set configuration value using dot notation path.
83 @param path Dot-separated path to configuration value
84 @param value Value to set
85 """
86 keys = path.split('.')
87 current = self._config
88
89 for key in keys[:-1]:
90 if key not in current:
91 current[key] = {}
92 current = current[key]
93
94 # Convert string values to appropriate types
95 if isinstance(value, str):
96 if value.lower() in ('true', 'false'):
97 value = value.lower() == 'true'
98 elif value.isdigit():
99 value = int(value)
100 elif value.replace('.', '', 1).isdigit():
101 value = float(value)
102
103 current[keys[-1]] = value

105 def get(self, key: str, default: Any = None, required: bool = False) -> Any:
106 """
107 Get a configuration value using dot notation. If required and missing, raise KeyError.
108 """
109 keys = key.split('.')
110 current = self._config
111 try:
112 for key_part in keys:
113 current = current[key_part]
114 if current is None and required:
115 raise KeyError(f"Required config key '{key}' is missing.")
116 return current
117 except (KeyError, TypeError):
118 if required:
119 raise KeyError(f"Required config key '{key}' is missing.")
120 return default
