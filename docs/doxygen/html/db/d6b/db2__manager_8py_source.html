155 def execute_query(self, query: str, params: Optional[Tuple] = None) -> List[Dict[str, Any]]:
156 """
157 Execute a SELECT query and return results.
158
159 @brief Execute SELECT query using DB2 CLI and return results as list of dictionaries.
160 @param query SQL query string
161 @param params Query parameters tuple (not supported in CLI mode)
162 @return List of dictionaries containing query results

164 """

166 raise Exception("Database connection not established")
167

169
170 try:
171 # Create a temporary file for the query
172 import tempfile
173 with tempfile.NamedTemporaryFile(mode='w', suffix='.sql', delete=False) as f:
174 f.write(query)
175 temp_file = f.name
176
177 # Execute query using DB2 CLI
178 result = self._execute_db2_command("import from", [temp_file, "of del", "insert into temp_table"])
179
180 # For SELECT queries, we need to use a different approach
181 # Export results to CSV and parse them
182 csv_result = self._execute_db2_command("export to", ["/tmp/query_result.csv", "of del", "select * from", "(" + query + ")"])
183
184 if csv_result['success']:
185 # Parse CSV results
186 results = self._parse_csv_results("/tmp/query_result.csv")
187
188 duration = time.time() - start_time
189 if self.logger:
190 self.logger.log_database_operation(
191 operation="SELECT",
192 query=query,
193 duration=duration,
194 rows_affected=len(results)
195 )
196
197 return results
198 else:
199 raise Exception(f"Query execution failed: {csv_result['stderr']}")
200
201 except Exception as e:
202 duration = time.time() - start_time
203 if self.logger:
204 self.logger.logger.error(f"Query execution error: {str(e)}")
205 self.logger.log_database_operation(
206 operation="SELECT",
207 query=query,
208 duration=duration,
209 rows_affected=0
210 )
211 raise
212
213 finally:
214 # Clean up temporary files

224 def _parse_csv_results(self, csv_file: str) -> List[Dict[str, Any]]:
225 """
226 Parse CSV results from DB2 export.
227
228 @brief Parse CSV file containing query results.
229 @param csv_file Path to CSV file
230 @return List of dictionaries containing parsed results
231 """
232 results = []
233 try:
234 with open(csv_file, 'r', encoding='utf-8') as f:
235 # Read first line to get column names
236 reader = csv.reader(f)
237 columns = next(reader, [])
238
239 # Read data rows
240 for row in reader:
241 if row: # Skip empty rows
242 result_dict = {}
243 for i, value in enumerate(row):
244 if i < len(columns):
245 result_dict[columns[i]] = value
246 results.append(result_dict)
247 except Exception as e:
248 if self.logger:
249 self.logger.logger.error(f"Error parsing CSV results: {str(e)}")
250
251 return results

315 def execute_batch(self, queries: List[str], params_list: Optional[List[Tuple]] = None) -> List[int]:
316 """
317 Execute multiple queries in a batch.
318
319 @brief Execute multiple queries using DB2 CLI in batch mode.
320 @param queries List of SQL query strings
321 @param params_list List of parameter tuples for each query (not supported in CLI mode)
322 @return List of affected rows for each query
323 @throws Exception If batch execution fails
324 """
325 if not self.is_connected():
326 raise Exception("Database connection not established")
327
328 results = []
329
330 try:
331 # Begin transaction
332 self._execute_db2_command("begin transaction")
333
334 for i, query in enumerate(queries):
335 rows_affected = self.execute_update(query, None)
336 results.append(rows_affected)
337
338 # Commit transaction
339 self._execute_db2_command("commit")
340 if self.logger:
341 self.logger.logger.info(f"Batch execution completed: {len(queries)} queries")
342
343 except Exception as e:
344 # Rollback transaction
345 self._execute_db2_command("rollback")
346 if self.logger:
347 self.logger.logger.error(f"Batch execution error: {str(e)}")
348 raise
349
350 return results
