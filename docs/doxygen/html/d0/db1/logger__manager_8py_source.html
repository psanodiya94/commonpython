28 def format(self, record: logging.LogRecord) -> str:
29 """
30 Format log record as JSON.
31
32 @brief Convert log record to JSON format.
33 @param record LogRecord instance to format
34 @return JSON string representation of the log record
35 """
36 log_entry = {
37 'timestamp': datetime.fromtimestamp(record.created).isoformat(),
38 'level': record.levelname,
39 'logger': record.name,
40 'message': record.getMessage(),
41 'module': record.module,
42 'function': record.funcName,
43 'line': record.lineno
44 }
45
46 # Add exception info if present
47 if record.exc_info:
48 log_entry['exception'] = self.formatException(record.exc_info)
49
50 # Add extra fields
51 for key, value in record.__dict__.items():
52 if key not in ['name', 'msg', 'args', 'levelname', 'levelno', 'pathname',
53 'filename', 'module', 'exc_info', 'exc_text', 'stack_info',
54 'lineno', 'funcName', 'created', 'msecs', 'relativeCreated',
55 'thread', 'threadName', 'processName', 'process', 'getMessage']:
56 log_entry[key] = value
57
58 return json.dumps(log_entry, default=str)

102 def _setup_console_handler(self) -> None:
103 """
104 Setup console handler with colored output if enabled.
105
106 @brief Configure console handler with optional colored output.

108 if not self.config.get('console', True):
109 return

122 self.logger.addHandler(console_handler)

124 def _setup_file_handler(self) -> None:
125 """
126 Setup file handler with rotation.
127
128 @brief Configure file handler with log rotation using standard Python modules.
129 """
130 log_file = self.config.get('file')
131 log_dir = self.config.get('dir')
132 if not log_file:
133 return
134 # If log_dir is specified, use it for log file location
135 if log_dir:
136 log_path = Path(log_dir) / Path(log_file).name
137 else:
138 log_path = Path(log_file)
139 log_path.parent.mkdir(parents=True, exist_ok=True)
140 max_bytes = self.config.get('max_size', 10 * 1024 * 1024) # 10MB
141 backup_count = self.config.get('backup_count', 5)
142 file_handler = logging.handlers.RotatingFileHandler(
143 str(log_path), maxBytes=max_bytes, backupCount=backup_count
144 )
145 file_handler.setLevel(logging.DEBUG)
146 if self.config.get('json_format', True):
147 file_formatter = JSONFormatter()
148 else:
149 file_formatter = logging.Formatter(
150 self.config.get('format', '%(asctime)s - %(name)s - %(levelname)s - %(message)s')
151 )
152 file_handler.setFormatter(file_formatter)
153 self.logger.addHandler(file_handler)

250class ColoredFormatter(logging.Formatter):
251 """
252 Colored formatter for console output.
253
254 @brief Provides colored console output using ANSI escape codes.
255 """
256
257 COLORS = {
258 'DEBUG': '\033[36m', # Cyan
259 'INFO': '\033[32m', # Green
260 'WARNING': '\033[33m', # Yellow
261 'ERROR': '\033[31m', # Red
262 'CRITICAL': '\033[35m', # Magenta
263 'RESET': '\033[0m' # Reset
264 }
265
266 def format(self, record: logging.LogRecord) -> str:
267 """
268 Format log record with colors.
269
270 @brief Add ANSI color codes to log level names.
271 @param record LogRecord instance to format
272 @return Colored string representation of the log record
273 """
274 # Get the original formatted message
275 message = super().format(record)
276
277 # Add color based on log level
278 level_name = record.levelname
279 if level_name in self.COLORS:
280 color = self.COLORS[level_name]
281 reset = self.COLORS['RESET']
282 message = message.replace(level_name, f"{color}{level_name}{reset}")
283
284 return message
